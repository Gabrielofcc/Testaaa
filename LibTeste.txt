local TweenService = game:GetService("TweenService")
local UserInputService = game:GetService("UserInputService")
local CoreGui = game:GetService("CoreGui")

local Library = {}

Library.Theme = {
    Bg = Color3.fromRGB(10,10,14),
    Second = Color3.fromRGB(18,18,24),
    Card = Color3.fromRGB(22,22,30),

    Accent = Color3.fromRGB(125,70,255),

    Stroke = Color3.fromRGB(45,45,58),

    Text = Color3.fromRGB(255,255,255),
    DarkText = Color3.fromRGB(170,170,170)
}

function Library:Tween(obj,props,time)
    TweenService:Create(
        obj,
        TweenInfo.new(
            time or 0.18,
            Enum.EasingStyle.Quart,
            Enum.EasingDirection.Out
        ),
        props
    ):Play()
end

function Library:Create(class,props)

    local obj = Instance.new(class)

    for i,v in pairs(props or {}) do
        obj[i] = v
    end

    return obj
end

function Library:Corner(obj,radius)

    self:Create("UICorner",{
        Parent = obj,
        CornerRadius = UDim.new(0,radius or 12)
    })
end

function Library:Stroke(obj,color)

    self:Create("UIStroke",{
        Parent = obj,
        Color = color or self.Theme.Stroke,
        Thickness = 1
    })
end

function Library:MakeWindow(cfg)

    local WindowName = cfg.Name or cfg.Title or "Asta Hub"

    local Gui = self:Create("ScreenGui",{
        Parent = CoreGui,
        Name = "BlackCloverUI",
        ResetOnSpawn = false
    })

    local Float = self:Create("TextButton",{
        Parent = Gui,
        Size = UDim2.new(0,50,0,50),
        Position = UDim2.new(0,20,0.5,0),
        BackgroundColor3 = self.Theme.Accent,
        Text = "✦",
        Font = Enum.Font.GothamBlack,
        TextSize = 22,
        TextColor3 = Color3.new(1,1,1),
        Visible = false
    })

    self:Corner(Float,999)

    local Main = self:Create("Frame",{
        Parent = Gui,
        Size = UDim2.new(0,520,0,340),
        Position = UDim2.new(0.5,-260,0.5,-170),
        BackgroundColor3 = self.Theme.Bg,
        BorderSizePixel = 0,
        Active = true
    })

    self:Corner(Main,18)
    self:Stroke(Main)

    local Topbar = self:Create("Frame",{
        Parent = Main,
        Size = UDim2.new(1,0,0,50),
        BackgroundTransparency = 1,
        Active = true
    })

    local Title = self:Create("TextLabel",{
        Parent = Topbar,
        BackgroundTransparency = 1,
        Position = UDim2.new(0,15,0,0),
        Size = UDim2.new(1,-100,1,0),
        Font = Enum.Font.GothamBold,
        Text = WindowName,
        TextSize = 18,
        TextColor3 = self.Theme.Text,
        TextXAlignment = Enum.TextXAlignment.Left
    })

    local Minimize = self:Create("TextButton",{
        Parent = Topbar,
        Size = UDim2.new(0,28,0,28),
        Position = UDim2.new(1,-70,0.5,-14),
        BackgroundColor3 = self.Theme.Card,
        Text = "-",
        Font = Enum.Font.GothamBold,
        TextSize = 20,
        TextColor3 = self.Theme.Text
    })

    self:Corner(Minimize,8)
    self:Stroke(Minimize)

    local Close = self:Create("TextButton",{
        Parent = Topbar,
        Size = UDim2.new(0,28,0,28),
        Position = UDim2.new(1,-35,0.5,-14),
        BackgroundColor3 = Color3.fromRGB(120,40,40),
        Text = "X",
        Font = Enum.Font.GothamBold,
        TextSize = 14,
        TextColor3 = Color3.new(1,1,1)
    })

    self:Corner(Close,8)

    local Sidebar = self:Create("Frame",{
        Parent = Main,
        Position = UDim2.new(0,10,0,60),
        Size = UDim2.new(0,135,1,-70),
        BackgroundColor3 = self.Theme.Second
    })

    self:Corner(Sidebar,14)

    local SideLayout = self:Create("UIListLayout",{
        Parent = Sidebar,
        Padding = UDim.new(0,6),
        HorizontalAlignment = Enum.HorizontalAlignment.Center
    })

    self:Create("UIPadding",{
        Parent = Sidebar,
        PaddingTop = UDim.new(0,10)
    })

    local Content = self:Create("Frame",{
        Parent = Main,
        Position = UDim2.new(0,155,0,60),
        Size = UDim2.new(1,-165,1,-70),
        BackgroundTransparency = 1
    })

    local dragging = false
    local dragStart
    local startPos

    Main.InputBegan:Connect(function(input)

        if input.UserInputType == Enum.UserInputType.MouseButton1 then
            dragging = true
            dragStart = input.Position
            startPos = Main.Position
        end
    end)

    UserInputService.InputChanged:Connect(function(input)

        if dragging and input.UserInputType == Enum.UserInputType.MouseMovement then

            local delta = input.Position - dragStart

            Main.Position = UDim2.new(
                startPos.X.Scale,
                startPos.X.Offset + delta.X,
                startPos.Y.Scale,
                startPos.Y.Offset + delta.Y
            )
        end
    end)

    UserInputService.InputEnded:Connect(function(input)

        if input.UserInputType == Enum.UserInputType.MouseButton1 then
            dragging = false
        end
    end)

    Minimize.MouseButton1Click:Connect(function()

        Main.Visible = false
        Float.Visible = true
    end)

    Float.MouseButton1Click:Connect(function()

        Main.Visible = true
        Float.Visible = false
    end)

    Close.MouseButton1Click:Connect(function()
        Gui:Destroy()
    end)

    local Window = {}
    local FirstTab = nil

    function Window:AddMinimizeButton()
        return true
    end

    function Window:MakeTab(cfg)

        local Name

        if typeof(cfg) == "table" then
            Name = cfg.Name or cfg[1] or "Tab"
        else
            Name = tostring(cfg)
        end

        local TabButton = Library:Create("TextButton",{
            Parent = Sidebar,
            Size = UDim2.new(1,-12,0,34),
            BackgroundColor3 = Library.Theme.Bg,
            Text = Name,
            Font = Enum.Font.GothamBold,
            TextSize = 13,
            TextColor3 = Library.Theme.DarkText,
            AutoButtonColor = false
        })

        Library:Corner(TabButton,10)
        Library:Stroke(TabButton)

        local Page = Library:Create("ScrollingFrame",{
            Parent = Content,
            Size = UDim2.new(1,0,1,0),
            BackgroundTransparency = 1,
            BorderSizePixel = 0,
            ScrollBarThickness = 3,
            Visible = false,
            CanvasSize = UDim2.new(0,0,0,0)
        })

        local Layout = Library:Create("UIListLayout",{
            Parent = Page,
            Padding = UDim.new(0,8)
        })

        Layout:GetPropertyChangedSignal("AbsoluteContentSize"):Connect(function()

            Page.CanvasSize = UDim2.new(
                0,
                0,
                0,
                Layout.AbsoluteContentSize.Y + 15
            )
        end)

        TabButton.MouseButton1Click:Connect(function()

            for _,v in pairs(Content:GetChildren()) do
                if v:IsA("ScrollingFrame") then
                    v.Visible = false
                end
            end

            for _,v in pairs(Sidebar:GetChildren()) do
                if v:IsA("TextButton") then

                    Library:Tween(v,{
                        BackgroundColor3 = Library.Theme.Bg,
                        TextColor3 = Library.Theme.DarkText
                    })
                end
            end

            Page.Visible = true

            Library:Tween(TabButton,{
                BackgroundColor3 = Library.Theme.Accent,
                TextColor3 = Color3.new(1,1,1)
            })
        end)

        if not FirstTab then

            FirstTab = TabButton

            task.wait()

            Page.Visible = true

            TabButton.BackgroundColor3 = Library.Theme.Accent
            TabButton.TextColor3 = Color3.new(1,1,1)
        end

        local Tab = {}

        function Tab:AddSection(cfg)

            local Label = Library:Create("TextLabel",{
                Parent = Page,
                Size = UDim2.new(1,-4,0,26),
                BackgroundTransparency = 1,
                Font = Enum.Font.GothamBold,
                Text = cfg.Name or cfg.Title or "Section",
                TextSize = 15,
                TextColor3 = Library.Theme.Accent,
                TextXAlignment = Enum.TextXAlignment.Left
            })

            return Label
        end

        function Tab:AddParagraph(cfg)

            local Frame = Library:Create("Frame",{
                Parent = Page,
                Size = UDim2.new(1,-4,0,65),
                BackgroundColor3 = Library.Theme.Card
            })

            Library:Corner(Frame,12)
            Library:Stroke(Frame)

            local Title = Library:Create("TextLabel",{
                Parent = Frame,
                BackgroundTransparency = 1,
                Position = UDim2.new(0,12,0,8),
                Size = UDim2.new(1,-24,0,18),
                Font = Enum.Font.GothamBold,
                Text = cfg.Title or cfg.Name or "Paragraph",
                TextSize = 14,
                TextColor3 = Library.Theme.Text,
                TextXAlignment = Enum.TextXAlignment.Left
            })

            local Desc = Library:Create("TextLabel",{
                Parent = Frame,
                BackgroundTransparency = 1,
                Position = UDim2.new(0,12,0,28),
                Size = UDim2.new(1,-24,0,30),
                Font = Enum.Font.Gotham,
                Text = cfg.Content or cfg.Text or "...",
                TextWrapped = true,
                TextSize = 12,
                TextColor3 = Library.Theme.DarkText,
                TextXAlignment = Enum.TextXAlignment.Left,
                TextYAlignment = Enum.TextYAlignment.Top
            })

            return Frame
        end

        function Tab:AddButton(cfg)

            local Button = Library:Create("TextButton",{
                Parent = Page,
                Size = UDim2.new(1,-4,0,42),
                BackgroundColor3 = Library.Theme.Card,
                Text = "",
                AutoButtonColor = false
            })

            Library:Corner(Button,12)
            Library:Stroke(Button)

            local Label = Library:Create("TextLabel",{
                Parent = Button,
                BackgroundTransparency = 1,
                Position = UDim2.new(0,12,0,0),
                Size = UDim2.new(1,-12,1,0),
                Font = Enum.Font.GothamMedium,
                Text = cfg.Name or "Button",
                TextSize = 14,
                TextColor3 = Library.Theme.Text,
                TextXAlignment = Enum.TextXAlignment.Left
            })

            Button.MouseButton1Click:Connect(function()

                if cfg.Callback then
                    pcall(cfg.Callback)
                end
            end)

            return Button
        end

        function Tab:AddToggle(cfg)

            local Enabled = cfg.Default or false

            local Toggle = Library:Create("TextButton",{
                Parent = Page,
                Size = UDim2.new(1,-4,0,42),
                BackgroundColor3 = Library.Theme.Card,
                Text = ""
            })

            Library:Corner(Toggle,12)
            Library:Stroke(Toggle)

            local Label = Library:Create("TextLabel",{
                Parent = Toggle,
                BackgroundTransparency = 1,
                Position = UDim2.new(0,12,0,0),
                Size = UDim2.new(1,-60,1,0),
                Font = Enum.Font.GothamMedium,
                Text = cfg.Name or "Toggle",
                TextSize = 14,
                TextColor3 = Library.Theme.Text,
                TextXAlignment = Enum.TextXAlignment.Left
            })

            local Indicator = Library:Create("Frame",{
                Parent = Toggle,
                Size = UDim2.new(0,22,0,22),
                Position = UDim2.new(1,-34,0.5,-11),
                BackgroundColor3 = Enabled and Library.Theme.Accent or Color3.fromRGB(40,40,40)
            })

            Library:Corner(Indicator,999)

            Toggle.MouseButton1Click:Connect(function()

                Enabled = not Enabled

                Library:Tween(Indicator,{
                    BackgroundColor3 = Enabled and Library.Theme.Accent or Color3.fromRGB(40,40,40)
                })

                if cfg.Callback then
                    pcall(cfg.Callback,Enabled)
                end
            end)

            return Toggle
        end

        function Tab:AddTextbox(cfg)

            local BoxFrame = Library:Create("Frame",{
                Parent = Page,
                Size = UDim2.new(1,-4,0,42),
                BackgroundColor3 = Library.Theme.Card
            })

            Library:Corner(BoxFrame,12)
            Library:Stroke(BoxFrame)

            local Box = Library:Create("TextBox",{
                Parent = BoxFrame,
                BackgroundTransparency = 1,
                Size = UDim2.new(1,-20,1,0),
                Position = UDim2.new(0,10,0,0),
                Font = Enum.Font.Gotham,
                PlaceholderText = cfg.Placeholder or cfg.Name or "Textbox",
                Text = "",
                TextSize = 14,
                TextColor3 = Library.Theme.Text
            })

            Box.FocusLost:Connect(function()

                if cfg.Callback then
                    pcall(cfg.Callback,Box.Text)
                end
            end)

            return Box
        end

        return Tab
    end

    return Window
end

return Library
