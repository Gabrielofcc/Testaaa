--// BLACK CLOVER MODERN UI (FULL FIXED + DRAG BORDER SUPPORT)

local TweenService = game:GetService("TweenService")
local UserInputService = game:GetService("UserInputService")
local CoreGui = game:GetService("CoreGui")

local ParentUI = gethui and gethui() or CoreGui

local Library = {}

Library.Theme = {
    Bg = Color3.fromRGB(10,10,14),
    Second = Color3.fromRGB(18,18,24),
    Card = Color3.fromRGB(22,22,30),

    Accent = Color3.fromRGB(125,70,255),

    Stroke = Color3.fromRGB(45,45,58),

    Text = Color3.fromRGB(255,255,255),
    DarkText = Color3.fromRGB(170,170,170)
}

function Library:Tween(obj,props,time)

    TweenService:Create(
        obj,
        TweenInfo.new(
            time or 0.18,
            Enum.EasingStyle.Quart,
            Enum.EasingDirection.Out
        ),
        props
    ):Play()
end

function Library:Create(class,props)

    local obj = Instance.new(class)

    for i,v in pairs(props or {}) do
        obj[i] = v
    end

    return obj
end

function Library:Corner(obj,radius)

    self:Create("UICorner",{
        Parent = obj,
        CornerRadius = UDim.new(0,radius or 12)
    })
end

function Library:Stroke(obj,color)

    self:Create("UIStroke",{
        Parent = obj,
        Color = color or self.Theme.Stroke,
        Thickness = 1
    })
end

function Library:MakeWindow(cfg)

    cfg = cfg or {}

    local WindowName = cfg.Name or cfg.Title or "Asta Hub"

    local Gui = self:Create("ScreenGui",{
        Parent = ParentUI,
        Name = "BlackCloverUI",
        ResetOnSpawn = false
    })

    -- FLOAT BUTTON

    local Float = self:Create("TextButton",{
        Parent = Gui,
        Size = UDim2.new(0,46,0,46),
        Position = UDim2.new(0,20,0.5,0),
        BackgroundColor3 = self.Theme.Accent,
        Text = "✦",
        Font = Enum.Font.GothamBlack,
        TextSize = 20,
        TextColor3 = Color3.new(1,1,1),
        Visible = false,
        AutoButtonColor = false
    })

    self:Corner(Float,999)

    -- MAIN

    local Main = self:Create("Frame",{
        Parent = Gui,
        Size = UDim2.new(0,470,0,330),
        Position = UDim2.new(0.5,-235,0.5,-165),
        BackgroundColor3 = self.Theme.Bg,
        BorderSizePixel = 0,
        Active = true
    })

    self:Corner(Main,16)
    self:Stroke(Main)

    -- GLOW

    local Glow = self:Create("ImageLabel",{
        Parent = Main,
        BackgroundTransparency = 1,
        Image = "rbxassetid://5028857084",
        ScaleType = Enum.ScaleType.Slice,
        SliceCenter = Rect.new(24,24,276,276),
        ImageColor3 = self.Theme.Accent,
        ImageTransparency = 0.5,
        Size = UDim2.new(1,36,1,36),
        Position = UDim2.new(0,-18,0,-18),
        ZIndex = 0
    })

    -- TOPBAR

    local Topbar = self:Create("Frame",{
        Parent = Main,
        Size = UDim2.new(1,0,0,45),
        BackgroundTransparency = 1,
        Active = true
    })

    local Title = self:Create("TextLabel",{
        Parent = Topbar,
        BackgroundTransparency = 1,
        Position = UDim2.new(0,14,0,0),
        Size = UDim2.new(1,-100,1,0),
        Font = Enum.Font.GothamBold,
        Text = WindowName,
        TextSize = 16,
        TextColor3 = self.Theme.Text,
        TextXAlignment = Enum.TextXAlignment.Left
    })

    -- MINIMIZE

    local Minimize = self:Create("TextButton",{
        Parent = Topbar,
        Size = UDim2.new(0,24,0,24),
        Position = UDim2.new(1,-62,0.5,-12),
        BackgroundColor3 = self.Theme.Card,
        Text = "-",
        Font = Enum.Font.GothamBold,
        TextSize = 18,
        TextColor3 = self.Theme.Text,
        AutoButtonColor = false
    })

    self:Corner(Minimize,8)
    self:Stroke(Minimize)

    -- CLOSE

    local Close = self:Create("TextButton",{
        Parent = Topbar,
        Size = UDim2.new(0,24,0,24),
        Position = UDim2.new(1,-32,0.5,-12),
        BackgroundColor3 = Color3.fromRGB(120,40,40),
        Text = "X",
        Font = Enum.Font.GothamBold,
        TextSize = 13,
        TextColor3 = Color3.new(1,1,1),
        AutoButtonColor = false
    })

    self:Corner(Close,8)

    -- SIDEBAR

    local Sidebar = self:Create("Frame",{
        Parent = Main,
        Position = UDim2.new(0,10,0,52),
        Size = UDim2.new(0,120,1,-62),
        BackgroundColor3 = self.Theme.Second,
        BorderSizePixel = 0
    })

    self:Corner(Sidebar,12)

    local SideLayout = self:Create("UIListLayout",{
        Parent = Sidebar,
        Padding = UDim.new(0,8),
        HorizontalAlignment = Enum.HorizontalAlignment.Center,
        SortOrder = Enum.SortOrder.LayoutOrder
    })

    self:Create("UIPadding",{
        Parent = Sidebar,
        PaddingTop = UDim.new(0,8)
    })

    -- CONTENT

    local Content = self:Create("Frame",{
        Parent = Main,
        Position = UDim2.new(0,140,0,52),
        Size = UDim2.new(1,-150,1,-62),
        BackgroundTransparency = 1
    })

    -- DRAG SUPPORT (BORDAS + TOPBAR)

    local dragging = false
    local dragStart
    local startPos

    local function StartDrag(input)

        dragging = true
        dragStart = input.Position
        startPos = Main.Position
    end

    local function StopDrag()

        dragging = false
    end

    local function Drag(input)

        if dragging then

            local delta = input.Position - dragStart

            Main.Position = UDim2.new(
                startPos.X.Scale,
                startPos.X.Offset + delta.X,
                startPos.Y.Scale,
                startPos.Y.Offset + delta.Y
            )
        end
    end

    Main.InputBegan:Connect(function(input)

        if input.UserInputType == Enum.UserInputType.MouseButton1
        or input.UserInputType == Enum.UserInputType.Touch then

            StartDrag(input)
        end
    end)

    Topbar.InputBegan:Connect(function(input)

        if input.UserInputType == Enum.UserInputType.MouseButton1
        or input.UserInputType == Enum.UserInputType.Touch then

            StartDrag(input)
        end
    end)

    UserInputService.InputChanged:Connect(function(input)

        if input.UserInputType == Enum.UserInputType.MouseMovement
        or input.UserInputType == Enum.UserInputType.Touch then

            Drag(input)
        end
    end)

    UserInputService.InputEnded:Connect(function(input)

        if input.UserInputType == Enum.UserInputType.MouseButton1
        or input.UserInputType == Enum.UserInputType.Touch then

            StopDrag()
        end
    end)

    -- BUTTONS

    Minimize.MouseButton1Click:Connect(function()

        Main.Visible = false
        Float.Visible = true
    end)

    Float.MouseButton1Click:Connect(function()

        Main.Visible = true
        Float.Visible = false
    end)

    Close.MouseButton1Click:Connect(function()

        Gui:Destroy()
    end)

    local Window = {}
    local FirstTab = false

    function Window:AddMinimizeButton(cfg)
        return true
    end

    function Window:MakeTab(cfg)

        local Name

        if typeof(cfg) == "table" then
            Name = cfg.Name or cfg.Title or cfg[1] or "Tab"
        else
            Name = tostring(cfg)
        end

        local TabButton = Library:Create("TextButton",{
            Parent = Sidebar,
            Size = UDim2.new(1,-10,0,42),
            BackgroundColor3 = Library.Theme.Bg,
            Text = Name,
            Font = Enum.Font.GothamBold,
            TextSize = 14,
            TextColor3 = Library.Theme.DarkText,
            AutoButtonColor = false
        })

        Library:Corner(TabButton,10)
        Library:Stroke(TabButton)

        local Page = Library:Create("ScrollingFrame",{
            Parent = Content,
            Size = UDim2.new(1,0,1,0),
            BackgroundTransparency = 1,
            BorderSizePixel = 0,
            ScrollBarThickness = 3,
            Visible = false,
            CanvasSize = UDim2.new(0,0,0,0),
            AutomaticCanvasSize = Enum.AutomaticSize.Y,
            ScrollingDirection = Enum.ScrollingDirection.Y,
            ScrollBarImageColor3 = Library.Theme.Accent
        })

        local Layout = Library:Create("UIListLayout",{
            Parent = Page,
            Padding = UDim.new(0,8),
            SortOrder = Enum.SortOrder.LayoutOrder,
            HorizontalAlignment = Enum.HorizontalAlignment.Center
        })

        TabButton.MouseButton1Click:Connect(function()

            for _,v in pairs(Content:GetChildren()) do
                if v:IsA("ScrollingFrame") then
                    v.Visible = false
                end
            end

            for _,v in pairs(Sidebar:GetChildren()) do
                if v:IsA("TextButton") then

                    Library:Tween(v,{
                        BackgroundColor3 = Library.Theme.Bg,
                        TextColor3 = Library.Theme.DarkText
                    })
                end
            end

            Page.Visible = true

            Library:Tween(TabButton,{
                BackgroundColor3 = Library.Theme.Accent,
                TextColor3 = Color3.new(1,1,1)
            })
        end)

        if not FirstTab then

            FirstTab = true

            Page.Visible = true

            TabButton.BackgroundColor3 = Library.Theme.Accent
            TabButton.TextColor3 = Color3.new(1,1,1)
        end

        local Tab = {}

        -- SECTION

        function Tab:AddSection(cfg)

            local text = cfg.Title or cfg.Name or cfg[1] or "Section"

            local Label = Library:Create("TextLabel",{
                Parent = Page,
                Size = UDim2.new(1,-6,0,30),
                BackgroundTransparency = 1,
                Font = Enum.Font.GothamBold,
                Text = tostring(text),
                TextSize = 16,
                TextColor3 = Library.Theme.Accent,
                TextXAlignment = Enum.TextXAlignment.Left
            })

            return Label
        end

        -- PARAGRAPH

        function Tab:AddParagraph(cfg)

            local title = cfg.Title or cfg[1] or "Title"
            local desc = cfg.Content or cfg.Desc or cfg[2] or "Description"

            local Frame = Library:Create("Frame",{
                Parent = Page,
                Size = UDim2.new(1,-2,0,80),
                AutomaticSize = Enum.AutomaticSize.Y,
                BackgroundColor3 = Library.Theme.Card,
                BorderSizePixel = 0
            })

            Library:Corner(Frame,10)
            Library:Stroke(Frame)

            Library:Create("UIPadding",{
                Parent = Frame,
                PaddingTop = UDim.new(0,10),
                PaddingBottom = UDim.new(0,10),
                PaddingLeft = UDim.new(0,12),
                PaddingRight = UDim.new(0,12)
            })

            Library:Create("UIListLayout",{
                Parent = Frame,
                Padding = UDim.new(0,4),
                SortOrder = Enum.SortOrder.LayoutOrder
            })

            local TitleLabel = Library:Create("TextLabel",{
                Parent = Frame,
                BackgroundTransparency = 1,
                Size = UDim2.new(1,0,0,20),
                AutomaticSize = Enum.AutomaticSize.Y,
                Font = Enum.Font.GothamBold,
                Text = tostring(title),
                TextSize = 14,
                TextWrapped = true,
                TextColor3 = Library.Theme.Text,
                TextXAlignment = Enum.TextXAlignment.Left
            })

            local DescLabel = Library:Create("TextLabel",{
                Parent = Frame,
                BackgroundTransparency = 1,
                Size = UDim2.new(1,0,0,18),
                AutomaticSize = Enum.AutomaticSize.Y,
                Font = Enum.Font.Gotham,
                Text = tostring(desc),
                TextSize = 13,
                TextWrapped = true,
                TextColor3 = Library.Theme.DarkText,
                TextXAlignment = Enum.TextXAlignment.Left
            })

            return Frame
        end

        -- BUTTON

        function Tab:AddButton(cfg)

            local Button = Library:Create("TextButton",{
                Parent = Page,
                Size = UDim2.new(1,-2,0,40),
                BackgroundColor3 = Library.Theme.Card,
                Text = "",
                AutoButtonColor = false
            })

            Library:Corner(Button,10)
            Library:Stroke(Button)

            local Label = Library:Create("TextLabel",{
                Parent = Button,
                BackgroundTransparency = 1,
                Position = UDim2.new(0,12,0,0),
                Size = UDim2.new(1,-12,1,0),
                Font = Enum.Font.GothamMedium,
                Text = cfg.Name or cfg.Title or cfg[1] or "Button",
                TextSize = 13,
                TextColor3 = Library.Theme.Text,
                TextXAlignment = Enum.TextXAlignment.Left
            })

            Button.MouseEnter:Connect(function()

                Library:Tween(Button,{
                    BackgroundColor3 = Color3.fromRGB(28,28,38)
                })
            end)

            Button.MouseLeave:Connect(function()

                Library:Tween(Button,{
                    BackgroundColor3 = Library.Theme.Card
                })
            end)

            Button.MouseButton1Click:Connect(function()

                if cfg.Callback then
                    pcall(cfg.Callback)
                elseif cfg[2] then
                    pcall(cfg[2])
                end
            end)

            return Button
        end

        return Tab
    end

    return Window
end

return Library
