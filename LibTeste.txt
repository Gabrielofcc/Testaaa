--// BLACK CLOVER UI LIB - MODERN REMAKE
--// visual style inspirado em Black Clover
--// animações fluidas + glow + blur + transições suaves

local TweenService = game:GetService("TweenService")
local UserInputService = game:GetService("UserInputService")
local Players = game:GetService("Players")
local CoreGui = game:GetService("CoreGui")

local Player = Players.LocalPlayer

local Library = {}
Library.Theme = {
    Background = Color3.fromRGB(10,10,14),
    Secondary = Color3.fromRGB(16,16,22),
    Accent = Color3.fromRGB(120, 40, 255),
    Accent2 = Color3.fromRGB(180, 80, 255),
    Text = Color3.fromRGB(240,240,240),
    DarkText = Color3.fromRGB(140,140,140),
    Stroke = Color3.fromRGB(40,40,50)
}

function Library:Tween(obj, props, speed)
    TweenService:Create(
        obj,
        TweenInfo.new(
            speed or 0.25,
            Enum.EasingStyle.Quart,
            Enum.EasingDirection.Out
        ),
        props
    ):Play()
end

function Library:Create(class, props)
    local obj = Instance.new(class)

    for i,v in pairs(props or {}) do
        obj[i] = v
    end

    return obj
end

function Library:Window(title)
    local ScreenGui = self:Create("ScreenGui", {
        Name = "BlackCloverUI",
        Parent = CoreGui,
        ResetOnSpawn = false,
        ZIndexBehavior = Enum.ZIndexBehavior.Sibling
    })

    local Main = self:Create("Frame", {
        Parent = ScreenGui,
        Size = UDim2.new(0, 620, 0, 420),
        Position = UDim2.new(0.5,-310,0.5,-210),
        BackgroundColor3 = self.Theme.Background,
        BorderSizePixel = 0
    })

    local Corner = self:Create("UICorner", {
        Parent = Main,
        CornerRadius = UDim.new(0,18)
    })

    local Stroke = self:Create("UIStroke", {
        Parent = Main,
        Color = self.Theme.Stroke,
        Thickness = 1.2
    })

    local Glow = self:Create("ImageLabel", {
        Parent = Main,
        BackgroundTransparency = 1,
        Image = "rbxassetid://5028857084",
        ScaleType = Enum.ScaleType.Slice,
        SliceCenter = Rect.new(24,24,276,276),
        ImageTransparency = 0.25,
        ImageColor3 = self.Theme.Accent,
        Size = UDim2.new(1,40,1,40),
        Position = UDim2.new(0,-20,0,-20),
        ZIndex = 0
    })

    local Topbar = self:Create("Frame", {
        Parent = Main,
        Size = UDim2.new(1,0,0,60),
        BackgroundTransparency = 1
    })

    local Title = self:Create("TextLabel", {
        Parent = Topbar,
        BackgroundTransparency = 1,
        Position = UDim2.new(0,24,0,0),
        Size = UDim2.new(1,-24,1,0),
        Font = Enum.Font.GothamBlack,
        Text = title or "BLACK CLOVER",
        TextColor3 = self.Theme.Text,
        TextSize = 24,
        TextXAlignment = Enum.TextXAlignment.Left
    })

    local AccentLine = self:Create("Frame", {
        Parent = Topbar,
        AnchorPoint = Vector2.new(0,1),
        Position = UDim2.new(0,24,1,-6),
        Size = UDim2.new(0,120,0,3),
        BackgroundColor3 = self.Theme.Accent,
        BorderSizePixel = 0
    })

    self:Create("UICorner", {
        Parent = AccentLine,
        CornerRadius = UDim.new(1,0)
    })

    local Sidebar = self:Create("Frame", {
        Parent = Main,
        Position = UDim2.new(0,14,0,75),
        Size = UDim2.new(0,170,1,-90),
        BackgroundColor3 = self.Theme.Secondary,
        BorderSizePixel = 0
    })

    self:Create("UICorner", {
        Parent = Sidebar,
        CornerRadius = UDim.new(0,14)
    })

    local SideLayout = self:Create("UIListLayout", {
        Parent = Sidebar,
        Padding = UDim.new(0,8),
        HorizontalAlignment = Enum.HorizontalAlignment.Center,
        SortOrder = Enum.SortOrder.LayoutOrder
    })

    local Padding = self:Create("UIPadding", {
        Parent = Sidebar,
        PaddingTop = UDim.new(0,12)
    })

    local Container = self:Create("Frame", {
        Parent = Main,
        Position = UDim2.new(0,195,0,75),
        Size = UDim2.new(1,-210,1,-90),
        BackgroundTransparency = 1
    })

    -- draggable smooth
    local dragging = false
    local dragStart
    local startPos

    Topbar.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 then
            dragging = true
            dragStart = input.Position
            startPos = Main.Position
        end
    end)

    UserInputService.InputEnded:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 then
            dragging = false
        end
    end)

    UserInputService.InputChanged:Connect(function(input)
        if dragging and input.UserInputType == Enum.UserInputType.MouseMovement then
            local delta = input.Position - dragStart

            self:Tween(Main,{
                Position = UDim2.new(
                    startPos.X.Scale,
                    startPos.X.Offset + delta.X,
                    startPos.Y.Scale,
                    startPos.Y.Offset + delta.Y
                )
            },0.08)
        end
    end)

    local WindowFunctions = {}

    function WindowFunctions:Tab(name)
        local TabButton = Library:Create("TextButton", {
            Parent = Sidebar,
            Size = UDim2.new(1,-16,0,42),
            BackgroundColor3 = Library.Theme.Background,
            Text = "",
            AutoButtonColor = false
        })

        Library:Create("UICorner", {
            Parent = TabButton,
            CornerRadius = UDim.new(0,12)
        })

        local TabText = Library:Create("TextLabel", {
            Parent = TabButton,
            BackgroundTransparency = 1,
            Size = UDim2.new(1,0,1,0),
            Font = Enum.Font.GothamBold,
            Text = name,
            TextColor3 = Library.Theme.DarkText,
            TextSize = 14
        })

        local Page = Library:Create("ScrollingFrame", {
            Parent = Container,
            Size = UDim2.new(1,0,1,0),
            CanvasSize = UDim2.new(0,0,0,0),
            ScrollBarThickness = 0,
            Visible = false,
            BackgroundTransparency = 1
        })

        local Layout = Library:Create("UIListLayout", {
            Parent = Page,
            Padding = UDim.new(0,10)
        })

        Layout:GetPropertyChangedSignal("AbsoluteContentSize"):Connect(function()
            Page.CanvasSize = UDim2.new(0,0,0,Layout.AbsoluteContentSize.Y + 20)
        end)

        TabButton.MouseEnter:Connect(function()
            if not Page.Visible then
                Library:Tween(TabButton,{
                    BackgroundColor3 = Color3.fromRGB(22,22,30)
                },0.15)
            end
        end)

        TabButton.MouseLeave:Connect(function()
            if not Page.Visible then
                Library:Tween(TabButton,{
                    BackgroundColor3 = Library.Theme.Background
                },0.15)
            end
        end)

        TabButton.MouseButton1Click:Connect(function()

            for _,v in pairs(Container:GetChildren()) do
                if v:IsA("ScrollingFrame") then
                    v.Visible = false
                end
            end

            for _,v in pairs(Sidebar:GetChildren()) do
                if v:IsA("TextButton") then
                    Library:Tween(v,{
                        BackgroundColor3 = Library.Theme.Background
                    },0.2)

                    local txt = v:FindFirstChildOfClass("TextLabel")
                    if txt then
                        Library:Tween(txt,{
                            TextColor3 = Library.Theme.DarkText
                        },0.2)
                    end
                end
            end

            Page.Visible = true

            Library:Tween(TabButton,{
                BackgroundColor3 = Library.Theme.Accent
            },0.2)

            Library:Tween(TabText,{
                TextColor3 = Color3.new(1,1,1)
            },0.2)
        end)

        local TabFunctions = {}

        function TabFunctions:Button(text, callback)
            local Button = Library:Create("TextButton", {
                Parent = Page,
                Size = UDim2.new(1,-5,0,50),
                BackgroundColor3 = Library.Theme.Secondary,
                Text = "",
                AutoButtonColor = false
            })

            Library:Create("UICorner", {
                Parent = Button,
                CornerRadius = UDim.new(0,14)
            })

            local Stroke = Library:Create("UIStroke", {
                Parent = Button,
                Color = Library.Theme.Stroke
            })

            local Label = Library:Create("TextLabel", {
                Parent = Button,
                BackgroundTransparency = 1,
                Size = UDim2.new(1,-20,1,0),
                Position = UDim2.new(0,20,0,0),
                Font = Enum.Font.GothamBold,
                Text = text,
                TextColor3 = Library.Theme.Text,
                TextSize = 15,
                TextXAlignment = Enum.TextXAlignment.Left
            })

            Button.MouseEnter:Connect(function()
                Library:Tween(Button,{
                    BackgroundColor3 = Color3.fromRGB(26,26,36)
                },0.15)

                Library:Tween(Stroke,{
                    Color = Library.Theme.Accent
                },0.15)
            end)

            Button.MouseLeave:Connect(function()
                Library:Tween(Button,{
                    BackgroundColor3 = Library.Theme.Secondary
                },0.15)

                Library:Tween(Stroke,{
                    Color = Library.Theme.Stroke
                },0.15)
            end)

            Button.MouseButton1Down:Connect(function()
                Library:Tween(Button,{
                    Size = UDim2.new(1,-10,0,46)
                },0.07)
            end)

            Button.MouseButton1Up:Connect(function()
                Library:Tween(Button,{
                    Size = UDim2.new(1,-5,0,50)
                },0.07)

                pcall(callback)
            end)
        end

        return TabFunctions
    end

    return WindowFunctions
end

return Library
