--// ASTA MODERN UI LIB
--// Black Clover Inspired
--// Modern + Smooth + Clean
--// No particles

local TweenService = game:GetService("TweenService")
local UserInputService = game:GetService("UserInputService")
local CoreGui = game:GetService("CoreGui")

local Library = {}

Library.Theme = {
    Background = Color3.fromRGB(8,8,12),
    Secondary = Color3.fromRGB(14,14,20),
    Card = Color3.fromRGB(18,18,26),

    Accent = Color3.fromRGB(125, 65, 255),
    AccentDark = Color3.fromRGB(85, 45, 180),

    Stroke = Color3.fromRGB(40,40,50),

    Text = Color3.fromRGB(255,255,255),
    DarkText = Color3.fromRGB(170,170,170)
}

function Library:Tween(obj, props, speed)
    TweenService:Create(
        obj,
        TweenInfo.new(
            speed or 0.2,
            Enum.EasingStyle.Quart,
            Enum.EasingDirection.Out
        ),
        props
    ):Play()
end

function Library:Create(class, props)
    local obj = Instance.new(class)

    for i,v in pairs(props or {}) do
        obj[i] = v
    end

    return obj
end

function Library:MakeWindow(cfg)
    local Title = cfg.Name or "ASTA HUB"

    local Gui = self:Create("ScreenGui", {
        Parent = CoreGui,
        Name = "AstaModernUI",
        ResetOnSpawn = false,
        ZIndexBehavior = Enum.ZIndexBehavior.Sibling
    })

    local Main = self:Create("Frame", {
        Parent = Gui,
        Size = UDim2.new(0, 700, 0, 430),
        Position = UDim2.new(0.5,-350,0.5,-215),
        BackgroundColor3 = self.Theme.Background,
        BorderSizePixel = 0
    })

    self:Create("UICorner", {
        Parent = Main,
        CornerRadius = UDim.new(0,20)
    })

    self:Create("UIStroke", {
        Parent = Main,
        Color = self.Theme.Stroke,
        Thickness = 1.3
    })

    -- glow
    local Glow = self:Create("ImageLabel", {
        Parent = Main,
        BackgroundTransparency = 1,
        Image = "rbxassetid://5028857084",
        ScaleType = Enum.ScaleType.Slice,
        SliceCenter = Rect.new(24,24,276,276),
        ImageColor3 = self.Theme.Accent,
        ImageTransparency = 0.35,
        Size = UDim2.new(1,60,1,60),
        Position = UDim2.new(0,-30,0,-30),
        ZIndex = 0
    })

    -- topbar
    local Topbar = self:Create("Frame", {
        Parent = Main,
        Size = UDim2.new(1,0,0,65),
        BackgroundTransparency = 1
    })

    local HubTitle = self:Create("TextLabel", {
        Parent = Topbar,
        BackgroundTransparency = 1,
        Position = UDim2.new(0,25,0,0),
        Size = UDim2.new(1,-25,1,0),
        Font = Enum.Font.GothamBlack,
        Text = Title,
        TextSize = 25,
        TextColor3 = self.Theme.Text,
        TextXAlignment = Enum.TextXAlignment.Left
    })

    local AccentBar = self:Create("Frame", {
        Parent = Topbar,
        Position = UDim2.new(0,25,1,-8),
        Size = UDim2.new(0,140,0,4),
        BackgroundColor3 = self.Theme.Accent,
        BorderSizePixel = 0
    })

    self:Create("UICorner", {
        Parent = AccentBar,
        CornerRadius = UDim.new(1,0)
    })

    -- sidebar
    local Sidebar = self:Create("Frame", {
        Parent = Main,
        Position = UDim2.new(0,15,0,80),
        Size = UDim2.new(0,185,1,-95),
        BackgroundColor3 = self.Theme.Secondary,
        BorderSizePixel = 0
    })

    self:Create("UICorner", {
        Parent = Sidebar,
        CornerRadius = UDim.new(0,16)
    })

    local SideLayout = self:Create("UIListLayout", {
        Parent = Sidebar,
        Padding = UDim.new(0,8),
        HorizontalAlignment = Enum.HorizontalAlignment.Center,
        SortOrder = Enum.SortOrder.LayoutOrder
    })

    self:Create("UIPadding", {
        Parent = Sidebar,
        PaddingTop = UDim.new(0,12)
    })

    -- content
    local Content = self:Create("Frame", {
        Parent = Main,
        Position = UDim2.new(0,215,0,80),
        Size = UDim2.new(1,-230,1,-95),
        BackgroundTransparency = 1
    })

    -- drag smooth
    local dragging = false
    local dragStart
    local startPos

    Topbar.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 then
            dragging = true
            dragStart = input.Position
            startPos = Main.Position
        end
    end)

    UserInputService.InputEnded:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 then
            dragging = false
        end
    end)

    UserInputService.InputChanged:Connect(function(input)
        if dragging and input.UserInputType == Enum.UserInputType.MouseMovement then

            local delta = input.Position - dragStart

            Library:Tween(Main,{
                Position = UDim2.new(
                    startPos.X.Scale,
                    startPos.X.Offset + delta.X,
                    startPos.Y.Scale,
                    startPos.Y.Offset + delta.Y
                )
            },0.06)
        end
    end)

    local Window = {}

    function Window:MakeTab(cfg)
        local Name = cfg.Name or "Tab"

        local TabButton = Library:Create("TextButton", {
            Parent = Sidebar,
            Size = UDim2.new(1,-16,0,45),
            BackgroundColor3 = Library.Theme.Background,
            Text = "",
            AutoButtonColor = false
        })

        self:CreateUICorner(TabButton)

        local TabStroke = Library:Create("UIStroke", {
            Parent = TabButton,
            Color = Library.Theme.Stroke
        })

        local TabText = Library:Create("TextLabel", {
            Parent = TabButton,
            BackgroundTransparency = 1,
            Size = UDim2.new(1,0,1,0),
            Font = Enum.Font.GothamBold,
            Text = Name,
            TextSize = 14,
            TextColor3 = Library.Theme.DarkText
        })

        local Page = Library:Create("ScrollingFrame", {
            Parent = Content,
            Visible = false,
            Size = UDim2.new(1,0,1,0),
            ScrollBarThickness = 0,
            BackgroundTransparency = 1,
            CanvasSize = UDim2.new(0,0,0,0)
        })

        local Layout = Library:Create("UIListLayout", {
            Parent = Page,
            Padding = UDim.new(0,10)
        })

        Layout:GetPropertyChangedSignal("AbsoluteContentSize"):Connect(function()
            Page.CanvasSize = UDim2.new(0,0,0,Layout.AbsoluteContentSize.Y + 20)
        end)

        TabButton.MouseEnter:Connect(function()
            if not Page.Visible then
                Library:Tween(TabButton,{
                    BackgroundColor3 = Color3.fromRGB(20,20,28)
                })
            end
        end)

        TabButton.MouseLeave:Connect(function()
            if not Page.Visible then
                Library:Tween(TabButton,{
                    BackgroundColor3 = Library.Theme.Background
                })
            end
        end)

        TabButton.MouseButton1Click:Connect(function()

            for _,v in pairs(Content:GetChildren()) do
                if v:IsA("ScrollingFrame") then
                    v.Visible = false
                end
            end

            for _,v in pairs(Sidebar:GetChildren()) do
                if v:IsA("TextButton") then

                    Library:Tween(v,{
                        BackgroundColor3 = Library.Theme.Background
                    })

                    local s = v:FindFirstChildOfClass("UIStroke")
                    local t = v:FindFirstChildOfClass("TextLabel")

                    if s then
                        Library:Tween(s,{
                            Color = Library.Theme.Stroke
                        })
                    end

                    if t then
                        Library:Tween(t,{
                            TextColor3 = Library.Theme.DarkText
                        })
                    end
                end
            end

            Page.Visible = true

            Library:Tween(TabButton,{
                BackgroundColor3 = Library.Theme.AccentDark
            })

            Library:Tween(TabStroke,{
                Color = Library.Theme.Accent
            })

            Library:Tween(TabText,{
                TextColor3 = Color3.new(1,1,1)
            })
        end)

        local Tab = {}

        function Tab:AddButton(cfg)

            local Text = cfg.Name or "Button"

            local Button = Library:Create("TextButton", {
                Parent = Page,
                Size = UDim2.new(1,-5,0,55),
                BackgroundColor3 = Library.Theme.Card,
                Text = "",
                AutoButtonColor = false
            })

            Library:Create("UICorner", {
                Parent = Button,
                CornerRadius = UDim.new(0,16)
            })

            local Stroke = Library:Create("UIStroke", {
                Parent = Button,
                Color = Library.Theme.Stroke
            })

            local Label = Library:Create("TextLabel", {
                Parent = Button,
                BackgroundTransparency = 1,
                Position = UDim2.new(0,18,0,0),
                Size = UDim2.new(1,-18,1,0),
                Font = Enum.Font.GothamBold,
                Text = Text,
                TextSize = 15,
                TextColor3 = Library.Theme.Text,
                TextXAlignment = Enum.TextXAlignment.Left
            })

            Button.MouseEnter:Connect(function()

                Library:Tween(Button,{
                    BackgroundColor3 = Color3.fromRGB(25,25,35)
                })

                Library:Tween(Stroke,{
                    Color = Library.Theme.Accent
                })
            end)

            Button.MouseLeave:Connect(function()

                Library:Tween(Button,{
                    BackgroundColor3 = Library.Theme.Card
                })

                Library:Tween(Stroke,{
                    Color = Library.Theme.Stroke
                })
            end)

            Button.MouseButton1Down:Connect(function()

                Library:Tween(Button,{
                    Size = UDim2.new(1,-10,0,50)
                },0.08)
            end)

            Button.MouseButton1Up:Connect(function()

                Library:Tween(Button,{
                    Size = UDim2.new(1,-5,0,55)
                },0.08)

                pcall(cfg.Callback)
            end)
        end

        return Tab
    end

    return Window
end

function Library:CreateUICorner(obj)
    self:Create("UICorner", {
        Parent = obj,
        CornerRadius = UDim.new(0,14)
    })
end

return Library
